Qdrant can be configured using YAML files, environment variables, or command-line arguments.

## Configuration Hierarchy

Configuration is loaded in the following order (later sources override earlier ones):

1. Default configuration (embedded)
2. `config/config.yaml` (base config)
3. `config/{RUN_MODE}.yaml` (environment-specific: development, production)
4. `config/local.yaml` (local overrides, not tracked in git)
5. `/etc/qdrant/config.yaml` (Debian package installs)
6. Custom config via `--config-path`
7. Environment variables (`QDRANT__*`)

## Environment Variables

All YAML settings can be overridden with environment variables using the pattern:

```
QDRANT__<SECTION>__<SUBSECTION>__<KEY>=<VALUE>
```

Examples:

```bash
QDRANT__LOG_LEVEL=DEBUG
QDRANT__SERVICE__HTTP_PORT=6333
QDRANT__SERVICE__API_KEY=secret-key
QDRANT__CLUSTER__ENABLED=true
QDRANT__CLUSTER__P2P__PORT=6335
QDRANT__STORAGE__STORAGE_PATH=/data/qdrant
```

## Complete Configuration Reference

### Logging

```yaml
log_level: INFO  # DEBUG, INFO, WARN, ERROR

logger:
  # Logging format: text or json
  format: text
  
  # Write logs to disk
  on_disk:
    enabled: false
    log_file: /var/log/qdrant/qdrant.log
    log_level: INFO
    format: text
    buffer_size_bytes: 1024
```

Environment variables:

```bash
QDRANT__LOG_LEVEL=INFO
QDRANT__LOGGER__FORMAT=json
QDRANT__LOGGER__ON_DISK__ENABLED=true
```

### Service Configuration

```yaml
service:
  # Host to bind to (0.0.0.0 for all interfaces)
  host: 0.0.0.0
  
  # HTTP REST API port
  http_port: 6333
  
  # gRPC API port (null to disable)
  grpc_port: 6334
  
  # Separate metrics port (not protected by API key)
  metrics_port: null
  
  # Maximum request size in megabytes
  max_request_size_mb: 32
  
  # Number of parallel workers (0 = auto)
  max_workers: 0
  
  # Enable CORS for REST API
  enable_cors: true
  
  # Enable TLS/HTTPS
  enable_tls: false
  
  # Verify HTTPS client certificates
  verify_https_client_certificate: false
  
  # API key for authentication
  api_key: null
  
  # Alternative API key (for key rotation)
  alt_api_key: null
  
  # Read-only API key
  read_only_api_key: null
  
  # Enable JWT RBAC
  jwt_rbac: false
  
  # Hide JWT dashboard
  hide_jwt_dashboard: false
  
  # Static content directory (Web UI)
  static_content_dir: null
  
  # Enable serving static content
  enable_static_content: true
  
  # Slow query threshold in seconds
  slow_query_secs: null
  
  # Enable hardware utilization reporting
  hardware_reporting: false
  
  # Metrics prefix for Prometheus
  metrics_prefix: qdrant_
```

Environment variables:

```bash
QDRANT__SERVICE__HOST=0.0.0.0
QDRANT__SERVICE__HTTP_PORT=6333
QDRANT__SERVICE__GRPC_PORT=6334
QDRANT__SERVICE__API_KEY=your-secret-key
QDRANT__SERVICE__ENABLE_TLS=true
```

### Storage Configuration

```yaml
storage:
  # Where to store all data
  storage_path: ./storage
  
  # Where to store snapshots
  snapshots_path: ./snapshots
  
  # Snapshot storage configuration
  snapshots_config:
    snapshots_storage: local  # local or s3
    # s3_config:
    #   bucket: my-bucket
    #   region: us-east-1
    #   access_key: ACCESS_KEY
    #   secret_key: SECRET_KEY
  
  # Temporary files path
  temp_path: null
  
  # Store payloads on disk (saves RAM)
  on_disk_payload: true
  
  # Maximum concurrent updates (null = unlimited)
  update_concurrency: null
  
  # Node type: Normal or Listener
  node_type: Normal
  
  # WAL configuration
  wal:
    wal_capacity_mb: 32
    wal_segments_ahead: 0
  
  # Performance tuning
  performance:
    # Search threads (0 = auto)
    max_search_threads: 0
    
    # Optimizer CPU budget (0 = auto, negative = subtract from total)
    optimizer_cpu_budget: 0
    
    # Update rate limit (null = auto)
    update_rate_limit: null
    
    # Shard transfer limits
    incoming_shard_transfers_limit: 1
    outgoing_shard_transfers_limit: 1
    
    # Enable async scorer with io_uring (Linux only)
    async_scorer: false
    
    # Concurrent loading limits
    max_concurrent_collection_loads: 1
    max_concurrent_shard_loads: 1
    max_concurrent_segment_loads: 8
  
  # Optimizer configuration
  optimizers:
    deleted_threshold: 0.2
    vacuum_min_vector_number: 1000
    default_segment_number: 0  # 0 = auto
    max_segment_size_kb: null
    indexing_threshold_kb: 10000
    flush_interval_sec: 5
    max_optimization_threads: null
  
  # Force-override all collection optimizers
  # optimizers_overwrite:
  #   deleted_threshold: 0.2
  
  # HNSW index defaults
  hnsw_index:
    m: 16
    ef_construct: 100
    full_scan_threshold_kb: 10000
    max_indexing_threads: 0
    on_disk: false
    payload_m: null
  
  # Default shard transfer method
  shard_transfer_method: null  # null, stream_records, snapshot, or wal_delta
  
  # Default collection settings
  collection:
    replication_factor: 1
    write_consistency_factor: 1
    
    vectors:
      on_disk: null
    
    quantization: null
    
    # Strict mode configuration
    # strict_mode:
    #   enabled: false
    #   max_query_limit: null
    #   max_timeout: null
  
  # Maximum number of collections (null = unlimited)
  max_collections: null
```

Environment variables:

```bash
QDRANT__STORAGE__STORAGE_PATH=/data/storage
QDRANT__STORAGE__SNAPSHOTS_PATH=/data/snapshots
QDRANT__STORAGE__ON_DISK_PAYLOAD=true
QDRANT__STORAGE__PERFORMANCE__MAX_SEARCH_THREADS=4
```

### Cluster Configuration

```yaml
cluster:
  # Enable distributed mode
  enabled: false
  
  # Peer ID (auto-assigned if not specified)
  peer_id: null
  
  # gRPC timeout for cluster operations
  grpc_timeout_ms: 5000
  
  # Connection timeout
  connection_timeout_ms: 5000
  
  # P2P configuration
  p2p:
    # Internal P2P port
    port: 6335
    
    # Connection pool size
    connection_pool_size: 4
    
    # Enable TLS for P2P
    enable_tls: false
  
  # Consensus configuration
  consensus:
    # Peer ping frequency
    tick_period_ms: 100
    
    # Bootstrap timeout
    bootstrap_timeout_sec: 15
    
    # Message timeout in ticks
    message_timeout_ticks: 10
    
    # WAL compaction threshold
    compact_wal_entries: 128
    
    # Max message queue size
    max_message_queue_size: 100
  
  # Enable resharding (experimental)
  resharding_enabled: false
```

Environment variables:

```bash
QDRANT__CLUSTER__ENABLED=true
QDRANT__CLUSTER__P2P__PORT=6335
QDRANT__CLUSTER__P2P__ENABLE_TLS=true
```

### TLS Configuration

```yaml
tls:
  # Server certificate
  cert: ./tls/cert.pem
  
  # Server private key
  key: ./tls/key.pem
  
  # CA certificate (for mTLS and P2P)
  ca_cert: ./tls/cacert.pem
  
  # Certificate reload TTL in seconds
  cert_ttl: 3600
```

Environment variables:

```bash
QDRANT__TLS__CERT=/path/to/cert.pem
QDRANT__TLS__KEY=/path/to/key.pem
QDRANT__TLS__CA_CERT=/path/to/ca.pem
QDRANT__TLS__CERT_TTL=3600
```

### Telemetry

```yaml
# Disable anonymous usage statistics
telemetry_disabled: false
```

Environment variable:

```bash
QDRANT__TELEMETRY_DISABLED=true
```

### Audit Logging

```yaml
audit:
  enabled: false
  dir: ./storage/audit
  rotation: daily  # daily, hourly, or size-based
  max_log_files: 7
  
  # Trust X-Forwarded-For header (only with trusted proxy)
  trust_forwarded_headers: false
```

Environment variables:

```bash
QDRANT__AUDIT__ENABLED=true
QDRANT__AUDIT__DIR=/var/log/qdrant/audit
```

### GPU Configuration

```yaml
gpu:
  # Enable GPU indexing
  indexing: false
  
  # Force half precision for f32
  force_half_precision: false
  
  # Number of GPU groups
  groups_count: 512
  
  # Filter devices by name
  device_filter: ""  # e.g., "nvidia"
  
  # Explicit device indices
  devices: null  # e.g., [0, 1]
  
  # Parallel indexing processes
  parallel_indexes: 1
  
  # Allow integrated GPUs
  allow_integrated: false
  
  # Allow emulated GPUs (for CI)
  allow_emulated: false
```

Environment variables:

```bash
QDRANT__GPU__INDEXING=true
QDRANT__GPU__DEVICE_FILTER=nvidia
```

## Command-Line Arguments

```bash
qdrant [OPTIONS]

Options:
  --uri <URI>
      URI of this peer for cluster communication
      [env: QDRANT_URI]
  
  --bootstrap <URI>
      URI of peer to bootstrap from
      [env: QDRANT_BOOTSTRAP]
  
  --config-path <PATH>
      Path to custom config file
  
  --snapshot <PATH:NAME>...
      Restore collection from snapshot
      Format: path/to/snapshot.snapshot:collection_name
  
  --storage-snapshot <PATH>
      Restore full storage from snapshot
  
  --force-snapshot
      Force snapshot recovery even if collections exist
  
  --disable-telemetry
      Disable telemetry collection
  
  --reinit
      Reinitialize consensus state
  
  --stacktrace
      Run stacktrace collector (debugging)
```

## Production Configuration Example

```yaml
log_level: WARN

logger:
  format: json
  on_disk:
    enabled: true
    log_file: /var/log/qdrant/qdrant.log

service:
  host: 0.0.0.0
  http_port: 6333
  grpc_port: 6334
  enable_tls: true
  verify_https_client_certificate: false
  api_key: ${QDRANT_API_KEY}  # From environment
  max_request_size_mb: 64
  max_workers: 0

storage:
  storage_path: /data/qdrant/storage
  snapshots_path: /data/qdrant/snapshots
  on_disk_payload: true
  
  performance:
    max_search_threads: 8
    optimizer_cpu_budget: 4
  
  collection:
    replication_factor: 3
    write_consistency_factor: 2

cluster:
  enabled: true
  p2p:
    port: 6335
    enable_tls: true

tls:
  cert: /etc/qdrant/tls/cert.pem
  key: /etc/qdrant/tls/key.pem
  ca_cert: /etc/qdrant/tls/ca.pem

audit:
  enabled: true
  dir: /var/log/qdrant/audit
  rotation: daily
  max_log_files: 30
```
