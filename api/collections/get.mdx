---
title: Get Collection Info
openapi: GET /collections/{name}
---

Retrieve detailed information about a specific collection.

## Path Parameters

<ParamField path="name" type="string" required>
  Name of the collection to retrieve
</ParamField>

## Example Request

```bash
curl -X GET http://localhost:6333/collections/my-collection
```

## Response

<ResponseField name="result" type="object">
  <ResponseField name="status" type="string">
    Collection status: `green`, `yellow`, or `red`
  </ResponseField>
  
  <ResponseField name="optimizer_status" type="string">
    Status of collection optimizers: `ok` or `error`
  </ResponseField>
  
  <ResponseField name="warnings" type="array">
    List of warnings related to the collection configuration
  </ResponseField>
  
  <ResponseField name="vectors_count" type="integer">
    Deprecated. Use `indexed_vectors_count` instead
  </ResponseField>
  
  <ResponseField name="indexed_vectors_count" type="integer">
    Approximate number of indexed vectors in the collection
  </ResponseField>
  
  <ResponseField name="points_count" type="integer">
    Approximate number of points (vectors + payloads) in the collection
  </ResponseField>
  
  <ResponseField name="segments_count" type="integer">
    Number of segments in the collection
  </ResponseField>
  
  <ResponseField name="config" type="object">
    Collection configuration
    
    <ResponseField name="params" type="object">
      Collection parameters
      
      <ResponseField name="vectors" type="object">
        Vector configuration (single or named)
      </ResponseField>
      
      <ResponseField name="shard_number" type="integer">
        Number of shards
      </ResponseField>
      
      <ResponseField name="replication_factor" type="integer">
        Number of replicas per shard
      </ResponseField>
      
      <ResponseField name="write_consistency_factor" type="integer">
        Write consistency factor
      </ResponseField>
      
      <ResponseField name="on_disk_payload" type="boolean">
        Whether payloads are stored on disk
      </ResponseField>
    </ResponseField>
    
    <ResponseField name="hnsw_config" type="object">
      HNSW index configuration
    </ResponseField>
    
    <ResponseField name="optimizer_config" type="object">
      Optimizers configuration
    </ResponseField>
    
    <ResponseField name="wal_config" type="object">
      Write-ahead log configuration
    </ResponseField>
    
    <ResponseField name="quantization_config" type="object" nullable>
      Quantization configuration, if enabled
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="payload_schema" type="object">
    Map of payload field names to their schema information
  </ResponseField>
  
  <ResponseField name="update_queue" type="object">
    Information about the update queue
    
    <ResponseField name="length" type="integer">
      Number of operations in the update queue
    </ResponseField>
  </ResponseField>
</ResponseField>
