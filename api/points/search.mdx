---
title: Search Points
api: POST /collections/{name}/points/search
---

Search for the nearest points to a given vector using vector similarity.

## Path Parameters

<ParamField path="name" type="string" required>
  Name of the collection to search in
</ParamField>

## Query Parameters

<ParamField query="consistency" type="string">
  Read consistency level for the search operation
</ParamField>

## Request Body

<ParamField body="vector" type="array | object" required>
  Vector to search for. Can be:
  - Dense vector: `[0.1, 0.2, 0.3, ...]`
  - Named vector: `{"name": "image", "vector": [0.1, 0.2]}`
</ParamField>

<ParamField body="limit" type="integer" default="10">
  Maximum number of results to return
</ParamField>

<ParamField body="offset" type="integer" default="0">
  Number of results to skip (for pagination)
</ParamField>

<ParamField body="filter" type="object">
  Filter conditions to narrow down the search space
  
  <Expandable title="Filter Examples">
    ```json
    {
      "must": [
        {"key": "city", "match": {"value": "Berlin"}}
      ]
    }
    ```
  </Expandable>
</ParamField>

<ParamField body="params" type="object">
  Search parameters
  
  <Expandable title="Search Params">
    <ParamField body="hnsw_ef" type="integer">
      Size of the dynamic candidate list for HNSW index. Higher values improve accuracy but reduce speed.
    </ParamField>
    
    <ParamField body="exact" type="boolean" default="false">
      If `true`, perform exact search instead of approximate. Slower but 100% accurate.
    </ParamField>
    
    <ParamField body="indexed_only" type="boolean" default="false">
      If `true`, search only in indexed vectors
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="score_threshold" type="number">
  Minimum score threshold for results. Points with lower scores will be excluded.
</ParamField>

<ParamField body="with_payload" type="boolean | array | object" default="false">
  Select which payload fields to return:
  - `true` - return all payload fields
  - `false` - don't return payload
  - `["field1", "field2"]` - return specific fields
  - `{"include": ["field1"]}` or `{"exclude": ["field2"]}`
</ParamField>

<ParamField body="with_vector" type="boolean | array" default="false">
  Whether to return vector data:
  - `true` - return all vectors
  - `false` - don't return vectors
  - `["vector_name"]` - return specific named vectors
</ParamField>

## Response

<ResponseField name="result" type="array">
  Array of scored points
  
  <Expandable title="Scored Point">
    <ResponseField name="id" type="integer | string">
      Point ID
    </ResponseField>
    
    <ResponseField name="score" type="number">
      Similarity score. Higher is more similar for cosine/dot, lower for Euclidean.
    </ResponseField>
    
    <ResponseField name="version" type="integer">
      Point version number
    </ResponseField>
    
    <ResponseField name="payload" type="object">
      Payload data (if requested)
    </ResponseField>
    
    <ResponseField name="vector" type="array | object">
      Vector data (if requested)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="status" type="string">
  Response status: `ok` or `error`
</ResponseField>

<ResponseField name="time" type="number">
  Execution time in seconds
</ResponseField>

## Examples

### Basic Search

```json
{
  "vector": [0.2, 0.1, 0.9, 0.7],
  "limit": 3
}
```

### Search with Filter

```json
{
  "vector": [0.2, 0.1, 0.9, 0.7],
  "limit": 5,
  "filter": {
    "must": [
      {
        "key": "city",
        "match": {
          "value": "London"
        }
      }
    ]
  }
}
```

### Search with Score Threshold

```json
{
  "vector": [0.2, 0.1, 0.9, 0.7],
  "limit": 10,
  "score_threshold": 0.5,
  "with_payload": true,
  "with_vector": false
}
```

### Search with Range Filter

```json
{
  "vector": [0.2, 0.1, 0.9, 0.7],
  "limit": 5,
  "filter": {
    "must": [
      {
        "key": "price",
        "range": {
          "gte": 10.0,
          "lte": 100.0
        }
      }
    ]
  },
  "params": {
    "hnsw_ef": 128,
    "exact": false
  }
}
```

### Named Vector Search

```json
{
  "vector": {
    "name": "image",
    "vector": [0.2, 0.1, 0.9, 0.7]
  },
  "limit": 3,
  "with_payload": ["name", "category"]
}
```

### Response Example

```json
{
  "result": [
    {
      "id": 1,
      "version": 3,
      "score": 0.95,
      "payload": {
        "city": "Berlin",
        "price": 15.99
      }
    },
    {
      "id": 5,
      "version": 2,
      "score": 0.87,
      "payload": {
        "city": "London",
        "price": 12.49
      }
    }
  ],
  "status": "ok",
  "time": 0.003
}
```
